function lasso(Y :: Matrix, =1.0, p=1)
    
    solver = SCSSolver(verbose=false)
    n, m = size(Y)
     = Variable(n, m)
    
    obj = 0.5 * square(norm(vec(Y - )))
    for i in 1:n
        for j in 1:m
            (j < m) ? (vec0 = abs([i, j] - [i, j+1])) : (vec0 = 0)
            (i < n) ? (vec1 = abs([i, j] - [i+1, j])) : (vec1 = 0)
            obj = obj +  * norm(vcat(vec0, vec1), p)
        end
    end
    
    prob = minimize(obj)
    solve!(prob, solver)
    return prob.optval, .value
end
